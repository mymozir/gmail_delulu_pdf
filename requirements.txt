# Основные компоненты сервера
fastapi                       # Современный и быстрый веб-фреймворк для создания API.
uvicorn                       # ASGI-сервер, который запускает FastAPI (наш "движок").
python-multipart              # Библиотека для обработки файлов, загруженных через HTTP POST-запросы.

# Библиотеки для анализа PDF
pypdf==4.0.1                  # Для высокоуровневого анализа PDF: извлечения метаданных, ссылок и страниц.
pdfminer.six                  # Для низкоуровневого анализа структуры PDF: поиска скрытых элементов и потоков (JS, OpenAction).

# Дополнительные инструменты
Pillow                        # Библиотека для работы с изображениями (понадобится для создания превью).

# Искусственный интеллект (AI)
google-genai                  # Библиотека для доступа к моделям Google Gemini (для генерации отчета).
